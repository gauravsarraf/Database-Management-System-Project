CREATE TABLE USERS(USER VARCHAR(30) PRIMARY KEY, PASS VARCHAR(15));
CREATE TABLE EMP_DATA(F_NAME VARCHAR(15),
		L_NAME VARCHAR(15),PER_ID VARCHAR(8) PRIMARY KEY,
		DOB DATE,USER VARCHAR(30) REFERENCES USERS(USER) ON DELETE CASCADE,
		REP_PER_ID VARCHAR(8),DESIGNATION VARCHAR(20),FREE_LEAVES INT(3));
CREATE TABLE DEPT_DATA (DEPT_ID INT(1) PRIMARY KEY, DEPT_NAME VARCHAR(5), PER_ID VARCHAR(8) REFERENCES EMP_DATA(PER_ID) ON DELETE CASCADE);
CREATE TABLE LEAVE_REQUEST ( LEAVE_ID INT(2) PRIMARY KEY, REP_EMP_ID VARCHAR(8), PER_ID VARCHAR(8) REFERENCES EMP_DATA(PER_ID) ON DELETE CASCADE, 
		REQ_DATE DATE, DAYS INT(2));
CREATE TABLE LEAVE_INFO (LEAVE_ID INT(2)REFERENCES LEAVE_REQUEST(LEAVE_ID) ON DELETE CASCADE, TOTAL_DAYS INT(2), 
		LEAVE_REASON VARCHAR(200), LEAVE_PRI INT(1), LEAVE_NAME VARCHAR(15), START_DATE DATE, END_DATE DATE, STATUS VARCHAR(20), REJECT_REASON VARCHAR(200));
